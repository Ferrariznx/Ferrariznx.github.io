# 组合总和

题目链接

解法:回溯算法 + 剪枝

思路：根据示例 1：输入: candidates = [2,3,6,7]，target = 7。

候选数组里有 2 ，如果找到了 7 - 2 = 5 的所有组合，再在之前加上 2 ，就是 7 的所有组合； 同理考虑 3，如果找到了 7 - 3 = 4 的所有组合，再在之前加上 3 ，就是 7 的所有组合，依次这样找下去； 上面的思路就可以画成下面的树形图。

var combinationSum = function(candidates, target) {
    
    let pathList=[]
    var findPath=function (value,path){
        if(value===0) return pathList.push(path.slice(0)); //slice的作用在于浅拷贝,防止多个path之间引用一个堆
        if(value<0) return;
        for(let i=0;i<candidates.length;i++){
            if(path.length===0 || (path.length>0 && path[path.length-1]>=candidates[i])){ //剪枝去重 如果能用2，3达到的效果用3，2也能达到，这里大于或者小于都是可以的
                var num=value-candidates[i]
                path.push(candidates[i])
                findPath(num,path)
                path.pop() //回溯 dfs完，让path出栈，让下一趟循环的path更干净
            }       
        }
    }
    findPath(target,[])
    return pathList;
};
